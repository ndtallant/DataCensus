def get_county_agencies(data, state, county):<br>

    '''<br>
    ori_list = [] <br>
    for agency in data[state]:<br>

            if data[state][agency]['county_name'] == county:<br>
            ori_list.append(data[state][agency])<br>
        except:<br>
            continue<br>
